\documentclass[<<fontsize>>]{article}
\usepackage{hyperref}
\usepackage{geometry}
\usepackage{enumitem}
\usepackage{underscore}
\usepackage[parfill]{parskip}
\usepackage{lmodern}
\usepackage[<<fontenc>>]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{fontspec}
\usepackage{sectsty}
\usepackage{multicol}

\def\resumeruler{\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}}
% Document configuration 

\setmainfont{<<mainfont>>}
\hypersetup{
  colorlinks = true,
  urlcolor=<< urlcolor >>,
  linkcolor=<< linkcolor >>,
  pdfauthor = {<< name >>},
  pdfkeywords = {<<# keywords >><<.>>, <</ keywords >>},
  pdftitle = {<< pdftitle >>},
  pdfsubject = {<< pdfsubject >>},
  pdfpagemode = UseNone
}
\geometry{
  body={6.5in, 9.0in},
  left=1.0in,
  top=1.0in
}

% Fix  for "! Undefined control sequence. <recently read> \tightlist",
% see: https://github.com/osener/markup.rocks/issues/4
% I have this issue with Pandoc 1.17.2
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Page number is top right, and it is possible to control the rest of
% the header.
\pagestyle{myheadings}
\markright{<< name >>}
\thispagestyle{empty}

% Custom section fonts
\sectionfont{\rmfamily\mdseries\Large}
\subsectionfont{\rmfamily\mdseries\bfseries\large}

% Section numbers or not (default)
<<# numbersections >>\setcounter{secnumdepth}{5}<</ numbersections >>
<<^ numbersections >>\setcounter{secnumdepth}{0}<</ numbersections >>

% By putting an empty \item[] at the start of the list, the list
% starts on a new line.
\setlist[itemize]{leftmargin=1em,before=\item[]}
%\setlist[itemize]{leftmargin=1em,label={--},before=\item[]} % label changes the bullet character
\setlist[description]{leftmargin=0em, style=sameline}
% Don't use monospace font for URLs
\urlstyle{same}


\begin{document}

% Place name at left
{\huge << name >>}

%%%%%%%% Info columns %%%%%%%%

<<# column >>
\bigskip

\begin{minipage}[t]{0.495\textwidth}
  <<#left>>
  \textbf{<<name>>} <<#link>>\href{<<to>>}{<<text>>}<</link>>\\
  <</left>>
\end{minipage} % Don't use empty lines after \end and the next \begin{minipage}.
\begin{minipage}[t]{0.495\textwidth}
  <<#right>>
  \textbf{<<name>>} <<#link>>\href{<<to>>}{<<text>>}<</link>>\\
  <</right>>
\end{minipage}
<</ column >>


%%%%%%%% Summary %%%%%%%%

<<# summary >>
\section{<<title>>}

<<body>>
<</ summary >>

%%%%%%%% Education %%%%%%%%

<<# education >>
<<# ruler >>\resumeruler<</ ruler >>
\section{<< title >>}

\begin{itemize}
  \tightlist
  <<# list >>
  \item \textbf{<< title >>}. << institution >><<# description >>. << . >><</ description >> <<# date >>(<< . >>)<</ date >>
  <</list>>
\end{itemize}
<</education>>

%%%%%%%% Skills %%%%%%%%

<<# skills>>
<<# ruler >>\resumeruler<</ ruler >>
\section{<<title>>}

\begin{multicols}{<<cols>>}
  \begin{itemize}
    <<#list>>
    \item \textbf{<<.>>}
    <</list>>
  \end{itemize}
\end{multicols}
<</skills>>

%%%%%%%% Experience %%%%%%%%

<<# experience >>
<<# ruler >>\resumeruler<</ ruler >>
\section{<< title >>}

<<# list >>
\subsection{<< company >>\hfill \small <<from>><<# to >> - << . >><</ to >>}
\textit{<<role>>. <<location>>}
\begin{itemize}
  \tightlist
  <<#description>>
  \item <<.>>
  <</description>>
\end{itemize}
<</list>>
<</experience>>


%%%%%%%% Formation %%%%%%%%

<<# formation >>
<<# ruler >>\resumeruler<</ ruler >>
\section{<< title >>}

\begin{itemize}
  \tightlist
  <<# list >>
    \item \textbf{<< title >>}. <<#link>>\href{<<to>>}{<<text>>}<</ link >> <<# date >>(<<.>>)<</date>>
  <</list>>
\end{itemize}
<</formation>>

%%%%%%%% Github %%%%%%%%

<<# github >>
<<# ruler >>\resumeruler<</ ruler >>
\section{<< title >>}

\begin{itemize}
  \tightlist
  <<# list >>
    \item \textbf{<< title >>}: <<summary>>. <<#link>>\href{<<to>>}{<<text>>}<</ link >>
  <</list>>
\end{itemize}
<</github>>

%%%%%%%% Other information %%%%%%%%

<<# other >>
<<# ruler >>\resumeruler<</ ruler >>
\section{<< title >>}

\begin{itemize}
  \tightlist
  <<# list >>
    \item \textbf{<< title >>}<<#summary>>. <<.>><</summary>>
  <</list>>
\end{itemize}
<</other>>

\end{document}
